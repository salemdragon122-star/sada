  <script>
    (function() {
      // ---------- Helper: toggle option boxes ----------
      function setupOptionBoxes(containerId) {
        const container = document.getElementById(containerId);
        if (!container) return;
        const boxes = container.querySelectorAll('.option-box');
        boxes.forEach(box => {
          box.addEventListener('click', function(e) {
            // Toggle selected class
            this.classList.toggle('selected');
          });
        });
      }

      setupOptionBoxes('foodOptions');
      setupOptionBoxes('activitiesOptions');

      // ---------- No button runaway ----------
      const noBtn = document.getElementById('noBtn');
      const card = document.querySelector('.card');

      function moveButtonAway(e) {
        // Get card boundaries to keep button inside
        const cardRect = card.getBoundingClientRect();
        const btnRect = noBtn.getBoundingClientRect();

        // Calculate random position inside the card (with padding)
        const padding = 20;
        const maxX = cardRect.width - btnRect.width - padding * 2;
        const maxY = cardRect.height - btnRect.height - padding * 2;

        // Get current position relative to card
        const currentX = noBtn.offsetLeft;
        const currentY = noBtn.offsetTop;

        let newX, newY;
        let attempts = 0;
        // Keep generating until position is significantly different (to avoid sticking)
        do {
          newX = Math.random() * maxX + padding;
          newY = Math.random() * maxY + padding;
          attempts++;
        } while (
          Math.abs(newX - currentX) < 30 && 
          Math.abs(newY - currentY) < 30 && 
          attempts < 20
        );

        noBtn.style.left = newX + 'px';
        noBtn.style.top = newY + 'px';
        noBtn.style.position = 'absolute';
      }

      // Initialize no button as absolute inside card (but card needs relative)
      card.style.position = 'relative';
      noBtn.style.position = 'absolute';
      noBtn.style.left = '60%'; // initial position
      noBtn.style.top = '20px';
      noBtn.style.transform = 'translateX(-50%)';

      // Move on mouseover (and also on click for safety)
      noBtn.addEventListener('mouseover', moveButtonAway);
      noBtn.addEventListener('click', function(e) {
        e.preventDefault();
        moveButtonAway(e);
      });

      // Also move when window resizes (to keep button inside)
      window.addEventListener('resize', function() {
        // reposition if out of bounds
        const cardRect = card.getBoundingClientRect();
        const btnRect = noBtn.getBoundingClientRect();
        let left = parseFloat(noBtn.style.left) || 0;
        let top = parseFloat(noBtn.style.top) || 0;
        if (left + btnRect.width > cardRect.width - 10) {
          noBtn.style.left = (cardRect.width - btnRect.width - 20) + 'px';
        }
        if (top + btnRect.height > cardRect.height - 10) {
          noBtn.style.top = (cardRect.height - btnRect.height - 20) + 'px';
        }
      });

      // ---------- Yes button ----------
      const yesBtn = document.getElementById('yesBtn');
      const responseMessage = document.getElementById('responseMessage');
      const heartsDisplay = document.getElementById('heartsDisplay');

      yesBtn.addEventListener('click', function() {
        // Get selected date
        const dateInput = document.getElementById('datePicker');
        const selectedDate = dateInput.value ? new Date(dateInput.value + 'T00:00:00') : null;
        const dateStr = selectedDate ? selectedDate.toLocaleDateString('en-US', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' }) : 'a day you choose';

        // Get selected food
        const foodBoxes = document.querySelectorAll('#foodOptions .option-box.selected');
        const foodList = Array.from(foodBoxes).map(box => box.innerText.trim()).join(', ') || 'something yummy';

        // Get selected activities
        const activityBoxes = document.querySelectorAll('#activitiesOptions .option-box.selected');
        const activityList = Array.from(activityBoxes).map(box => box.innerText.trim()).join(', ') || 'something fun';

        // Build lovely response
        let msg = `💕 Amazing! I can't wait for our date on <strong>${dateStr}</strong>!<br>`;
        msg += `We'll eat <strong>${foodList}</strong> and then <strong>${activityList}</strong>.<br>`;
        msg += `It's going to be absolutely magical! ✨💖`;

        responseMessage.innerHTML = msg;

        // Add floating hearts
        heartsDisplay.innerHTML = '💖💗💕💘💝';
        setTimeout(() => {
          heartsDisplay.innerHTML = '';
        }, 4000);

        // Disable yes button briefly for fun
        yesBtn.disabled = true;
        yesBtn.style.opacity = 0.6;
        setTimeout(() => {
          yesBtn.disabled = false;
          yesBtn.style.opacity = 1;
        }, 3000);
      });

      // ---------- Prevent no button from being clicked via accidental enter etc ----------
      noBtn.addEventListener('mousedown', function(e) {
        e.preventDefault();
        moveButtonAway(e);
      });

      // Also move on touchstart for mobile
      noBtn.addEventListener('touchstart', function(e) {
        e.preventDefault();
        moveButtonAway(e);
      });

      // Extra: move button to safe position on load
      setTimeout(() => {
        moveButtonAway({});
      }, 100);

    })();
  </script>
